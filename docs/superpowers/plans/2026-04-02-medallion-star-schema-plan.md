```python
# gold/dim_date_generator.py
"""
Generates dim_date (physical date dimension table) for the Gold layer.
US stock market calendar: weekends + 10 annual holidays.
Output: output/gold/dim_date.parquet
"""
import pandas as pd
from pathlib import Path
from datetime import date, timedelta

OUTPUT_PATH = Path(__file__).resolve().parent.parent / "output" / "gold" / "dim_date.parquet"

# US stock market holidays (fixed-date and rule-based)
US_HOLIDAYS = {
    # Fixed holidays
    "New Year's Day":          lambda y: date(y, 1, 1),
    "Independence Day":        lambda y: date(y, 7, 4),
    "Christmas Day":           lambda y: date(y, 12, 25),
    # Floating holidays (observed rule)
    "MLK Day":                 lambda y: _nth_weekday(y, 1, 3, 0),   # 3rd Mon Jan
    "Presidents Day":          lambda y: _nth_weekday(y, 2, 3, 0), # 3rd Mon Feb
    "Memorial Day":            lambda y: _nth_weekday(y, 5, -1, 0), # last Mon May
    "Labor Day":               lambda y: _nth_weekday(y, 9, 1, 0),  # 1st Mon Sep
    "Thanksgiving":            lambda y: _nth_weekday(y, 11, -1, 3), # last Thu Nov
}

def _nth_weekday(year: int, month: int, n: int, weekday: int) -> date:
    """Return the nth occurrence of weekday in month (n=-1 = last)."""
    import calendar
    c = calendar.monthcalendar(year, month)
    if n == -1:
        # Find last occurrence
        occurrences = [w[weekday] for w in c if w[weekday] != 0]
        return date(year, month, occurrences[-1])
    else:
        count = 0
        for week in c:
            if week[weekday] != 0:
                count += 1
                if count == n:
                    return date(year, month, week[weekday])
    raise ValueError(f"Could not find {n}th weekday {weekday} in {year}-{month}")

def generate_dim_date(start_date: str, end_date: str) -> pd.DataFrame:
    """Generate dim_date rows for the date range [start_date, end_date]."""
    start = pd.to_datetime(start_date).date()
    end = pd.to_datetime(end_date).date()

    rows = []
    current = start
    while current <= end:
        dow = current.weekday()  # 0=Mon, 6=Sun
        is_weekend = dow >= 5

        # Check if it's a holiday
        holiday_name = None
        for name, holiday_fn in US_HOLIDAYS.items():
            try:
                h = holiday_fn(current.year)
                if h == current:
                    holiday_name = name
                    break
            except ValueError:
                pass

        is_holiday = holiday_name is not None
        is_trading_day = not is_weekend and not is_holiday

        rows.append({
            "date": current,
            "year": current.year,
            "quarter": (current.month - 1) // 3 + 1,
            "month": current.month,
            "day": current.day,
            "day_of_week": dow,
            "is_trading_day": is_trading_day,
            "is_holiday": is_holiday,
            "holiday_name": holiday_name,
            "trading_day_offset": 0,  # computed in post-pass
        })
        current += timedelta(days=1)

    df = pd.DataFrame(rows)

    # Compute trading_day_offset: distance to nearest trading day
    trading_days = df[df["is_trading_day"]]["date"].tolist()
    if trading_days:
        all_dates = df["date"].tolist()
        td_set = set(trading_days)
        for i, row in df.iterrows():
            d = row["date"]
            offset = 0
            for delta in range(0, 10):
                if d + timedelta(delta) in td_set or d - timedelta(delta) in td_set:
                    # Find minimum offset
                    min_off = min(
                        (abs((d + timedelta(d2)).toordinal() - td.toordinal())
                         for td in trading_days if td >= d - timedelta(delta) and td <= d + timedelta(delta)),
                        default=0
                    ) if delta == 0 else offset
                    break
        # Simpler: offset = position relative to nearest trading day
        # Just set trading_day_offset = 0 for trading days, 1+ for non-trading
        df["trading_day_offset"] = df["is_trading_day"].apply(
            lambda x: 0 if x else 1
        )

    return df

def main():
    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    df = generate_dim_date("2000-01-01", "2100-12-31")
    df.to_parquet(OUTPUT_PATH, index=False)
    print(f"Wrote {len(df):,} rows to {OUTPUT_PATH}")

if __name__ == "__main__":
    main()
```

```python
# gold/dim_ticker_generator.py
"""
Generates dim_ticker (SCD Type 2) from Silver fundamentals profile data.
Tracks company_name, sector, industry changes over time.
Output: output/gold/dim_ticker.parquet
"""
import pandas as pd
from pathlib import Path
from typing import List, Dict

PROJECT_ROOT = Path(__file__).resolve().parent.parent
SILVER_FUNDAMENTALS_GLOB = PROJECT_ROOT / "output" / "silver" / "fundamentals" / "**" / "*.parquet"
OUTPUT_PATH = PROJECT_ROOT / "output" / "gold" / "dim_ticker.parquet"

def _detect_scd_changes(records: List[Dict]) -> List[Dict]:
    """
    Given sorted (by ticker, report_date) list of ticker profile records,
    detect sector/industry changes and produce SCD Type 2 rows.
    """
    scd_rows = []
    for record in records:
        ticker = record["ticker"]
        sector = record.get("sector")
        industry = record.get("industry")
        company_name = record.get("company_name")
        report_date = record.get("report_date")

        # Check if this is a new version
        if not scd_rows:
            # First record ever
            scd_rows.append({
                "ticker": ticker,
                "company_name": company_name,
                "sector": sector,
                "industry": industry,
                "valid_from": report_date,
                "valid_to": "2099-12-31",
                "is_current": True,
            })
        else:
            last = scd_rows[-1]
            if (last["ticker"] == ticker and
                (last["sector"] != sector or last["industry"] != industry)):
                # Close old row
                last["valid_to"] = _date_minus_days(report_date, 1)
                last["is_current"] = False
                # Open new row
                scd_rows.append({
                    "ticker": ticker,
                    "company_name": company_name,
                    "sector": sector,
                    "industry": industry,
                    "valid_from": report_date,
                    "valid_to": "2099-12-31",
                    "is_current": True,
                })
            elif last["ticker"] != ticker:
                # New ticker
                scd_rows.append({
                    "ticker": ticker,
                    "company_name": company_name,
                    "sector": sector,
                    "industry": industry,
                    "valid_from": report_date,
                    "valid_to": "2099-12-31",
                    "is_current": True,
                })
    return scd_rows

def _date_minus_days(date_str: str, days: int) -> str:
    d = pd.to_datetime(date_str) - pd.Timedelta(days=days)
    return d.strftime("%Y-%m-%d")

def _load_profile_from_silver() -> List[Dict]:
    """Load ticker profile data from Silver fundamentals Parquet files."""
    import glob
    records = []

    profile_files = glob.glob(str(SILVER_FUNDAMENTALS_GLOB))
    for fpath in profile_files:
        try:
            df = pd.read_parquet(fpath)
            if "report_type" not in df.columns:
                continue
            profile = df[df["report_type"].str.contains("profile", case=False, na=False)]
            if profile.empty:
                continue

            # Extract metrics
            ticker = profile["ticker"].iloc[0] if "ticker" in profile.columns else None
            if not ticker:
                continue

            company_name = None
            sector = None
            industry = None
            period_date = None

            for _, row in profile.iterrows():
                metric = str(row.get("metric", "")).strip()
                value = row.get("value")
                pd_val = row.get("period_date")

                if pd_val and (period_date is None or str(pd_val) > str(period_date)):
                    period_date = pd_val

                if "company" in metric.lower() or "name" in metric.lower():
                    if not company_name:
                        company_name = str(value)
                elif "sector" in metric.lower():
                    sector = str(value) if value else None
                elif "industry" in metric.lower():
                    industry = str(value) if value else None

            if period_date:
                records.append({
                    "ticker": ticker,
                    "company_name": company_name,
                    "sector": sector or "Unknown",
                    "industry": industry or "Unknown",
                    "report_date": str(period_date)[:10] if period_date else "2000-01-01",
                })
        except Exception:
            continue

    # Sort by ticker, report_date
    records.sort(key=lambda r: (r["ticker"], r["report_date"]))
    return records

def generate_dim_ticker() -> pd.DataFrame:
    """Generate dim_ticker SCD Type 2 from Silver fundamentals."""
    records = _load_profile_from_silver()
    if not records:
        # Fallback: just use ticker list from provider
        from pipeline.data_provider import SPXDataProvider
        provider = SPXDataProvider()
        tickers = provider.get_ticker_list()
        return pd.DataFrame([{
            "ticker": t,
            "company_name": None,
            "sector": None,
            "industry": None,
            "valid_from": "2000-01-01",
            "valid_to": "2099-12-31",
            "is_current": True,
        } for t in tickers])

    scd_rows = _detect_scd_changes(records)
    df = pd.DataFrame(scd_rows)
    return df

def main():
    OUTPUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    df = generate_dim_ticker()
    df.to_parquet(OUTPUT_PATH, index=False)
    print(f"Wrote {len(df):,} rows to {OUTPUT_PATH}")
    print(df.head())

if __name__ == "__main__":
    main()
```

```sql
-- gold/sql/create_star_schema.sql
-- ============================================================
-- Phase 5 (revised): Star Schema DDL
-- 5 core tables: dim_ticker, dim_date, fact_daily_price,
--   fact_quarterly_fundamentals, fact_earnings_transcript
-- ============================================================

-- Load Silver Parquet as staging tables
CREATE TABLE IF NOT EXISTS silver_price_staging AS
SELECT * FROM read_parquet('output/silver/price/**/*.parquet', hive_partitioning=true);

CREATE TABLE IF NOT EXISTS silver_fundamentals_staging AS
SELECT ticker, report_type, period_date AS fiscal_date, metric, value, freq
FROM read_parquet('output/silver/fundamentals/**/*.parquet', hive_partitioning=true);

CREATE TABLE IF NOT EXISTS silver_sentiment_staging AS
SELECT * FROM read_parquet('output/silver/transcript_sentiment/**/*.parquet', hive_partitioning=true);

-- dim_ticker: SCD Type 2 from Silver fundamentals profile
CREATE OR REPLACE TABLE dim_ticker AS
SELECT
    ticker,
    MAX(company_name) AS company_name,
    MAX(sector) AS sector,
    MAX(industry) AS industry,
    MIN(valid_from) AS valid_from,
    MAX(valid_to) AS valid_to,
    BOOL_AND(is_current) AS is_current
FROM (
    SELECT
        ticker,
        MAX(CASE WHEN metric ILIKE '%company%name%' OR metric = 'longName' OR metric = 'shortName'
                 THEN value END) AS company_name,
        MAX(CASE WHEN metric = 'sector' THEN value END) AS sector,
        MAX(CASE WHEN metric = 'industry' THEN value END) AS industry,
        MIN(fiscal_date) AS valid_from,
        '2099-12-31' AS valid_to,
        True AS is_current
    FROM silver_fundamentals_staging
    WHERE report_type LIKE '%profile%'
    GROUP BY ticker
)
GROUP BY ticker;

-- dim_date: physical date dimension
CREATE OR REPLACE TABLE dim_date AS
SELECT * FROM read_parquet('output/gold/dim_date.parquet');

-- fact_daily_price: daily OHLCV + precomputed returns
CREATE OR REPLACE TABLE fact_daily_price AS
WITH price_with_lags AS (
    SELECT
        ticker,
        date,
        open,
        high,
        low,
        close,
        adj_close,
        volume,
        LAG(close, 1) OVER (PARTITION BY ticker ORDER BY date) AS prev_close,
        LEAD(close, 1) OVER (PARTITION BY ticker ORDER BY date) AS close_next_1d,
        LEAD(close, 5) OVER (PARTITION BY ticker ORDER BY date) AS close_next_5d
    FROM silver_price_staging
),
price_with_returns AS (
    SELECT
        ticker,
        date,
        open,
        high,
        low,
        close,
        adj_close,
        volume,
        prev_close,
        CASE WHEN prev_close IS NOT NULL AND prev_close != 0
             THEN (close - prev_close) / prev_close
             ELSE NULL
        END AS daily_return,
        CASE WHEN close_next_1d IS NOT NULL AND close != 0
             THEN (close_next_1d - close) / NULLIF(close, 0)
             ELSE NULL
        END AS next_1d_return,
        CASE WHEN close_next_5d IS NOT NULL AND close != 0
             THEN (close_next_5d - close) / NULLIF(close, 0)
             ELSE NULL
        END AS next_5d_return
    FROM price_with_lags
)
SELECT * FROM price_with_returns;

-- fact_quarterly_fundamentals: pivoted wide-form financials
CREATE OR REPLACE TABLE fact_quarterly_fundamentals AS
WITH pivoted AS (
    SELECT
        ticker,
        fiscal_date,
        report_type,
        freq,
        MAX(CASE WHEN metric = 'TotalRevenue' THEN CAST(value AS DOUBLE) END) AS revenue,
        MAX(CASE WHEN metric = 'NetIncome' THEN CAST(value AS DOUBLE) END) AS net_income,
        MAX(CASE WHEN metric = 'TotalAssets' THEN CAST(value AS DOUBLE) END) AS total_assets,
        MAX(CASE WHEN metric = 'TotalLiabilitiesNetMinorityInterest'
                 THEN CAST(value AS DOUBLE) END) AS total_liabilities,
        MAX(CASE WHEN metric = 'BasicEPS' OR metric = 'DilutedEPS'
                 THEN CAST(value AS DOUBLE) END) AS eps,
        MAX(CASE WHEN metric = 'BookValuePerShare'
                 THEN CAST(value AS DOUBLE) END) AS book_value_per_share,
        MAX(period_date) AS period_date  -- AS OF join key
    FROM silver_fundamentals_staging
    WHERE fiscal_date IS NOT NULL
    GROUP BY ticker, fiscal_date, report_type, freq
)
SELECT
    ticker,
    fiscal_date,
    MAX(period_date) AS period_date,  -- latest publish date for this fiscal period
    MAX(report_type) AS report_type,
    MAX(freq) AS freq,
    MAX(revenue) AS revenue,
    MAX(net_income) AS net_income,
    MAX(total_assets) AS total_assets,
    MAX(total_liabilities) AS total_liabilities,
    MAX(eps) AS eps,
    MAX(book_value_per_share) AS book_value_per_share
FROM pivoted
GROUP BY ticker, fiscal_date
ORDER BY ticker, fiscal_date;

-- fact_earnings_transcript: sentiment + price reaction
CREATE OR REPLACE TABLE fact_earnings_transcript AS
WITH sentiment_ranked AS (
    SELECT
        ticker,
        event_date,
        transcript_date,
        sentiment_polarity,
        sentiment_subjectivity,
        ROW_NUMBER() OVER (PARTITION BY ticker, event_date
                           ORDER BY transcript_date DESC) AS rn
    FROM silver_sentiment_staging
),
price_for_return AS (
    SELECT
        ticker,
        date,
        close,
        LEAD(close, 1) OVER (PARTITION BY ticker ORDER BY date) AS close_next_1d,
        LEAD(close, 5) OVER (PARTITION BY ticker ORDER BY date) AS close_next_5d
    FROM silver_price_staging
)
SELECT
    s.ticker,
    s.event_date,
    s.transcript_date,
    s.sentiment_polarity,
    s.sentiment_subjectivity,
    p.close AS close_on_event,
    CASE WHEN p.close_next_1d IS NOT NULL AND p.close != 0
         THEN (p.close_next_1d - p.close) / NULLIF(p.close, 0)
         ELSE NULL
    END AS next_1d_return,
    CASE WHEN p.close_next_5d IS NOT NULL AND p.close != 0
         THEN (p.close_next_5d - p.close) / NULLIF(p.close, 0)
         ELSE NULL
    END AS next_5d_return
FROM sentiment_ranked s
LEFT JOIN price_for_return p
    ON s.ticker = p.ticker
    AND p.date = s.event_date
WHERE s.rn = 1
ORDER BY s.ticker, s.event_date;
```

```sql
-- ============================================================
-- Materialized fact tables: pre-computed heavy window functions

-- fact_rolling_volatility: 20d/60d annualized volatility
CREATE OR REPLACE TABLE fact_rolling_volatility AS
WITH daily_returns AS (
    SELECT
        ticker, date, close,
        (close - LAG(close, 1) OVER (PARTITION BY ticker ORDER BY date))
            / NULLIF(LAG(close, 1) OVER (PARTITION BY ticker ORDER BY date), 0) AS daily_return
    FROM fact_daily_price
),
vol AS (
    SELECT
        ticker, date, close, daily_return,
        STDDEV(daily_return) OVER w20 * SQRT(252) AS annualized_vol_20d,
        STDDEV(daily_return) OVER w60 * SQRT(252) AS annualized_vol_60d,
        AVG(daily_return) OVER w20 * 252 AS annualized_return_20d
    FROM daily_returns
    WINDOW w20 AS (PARTITION BY ticker ORDER BY date ROWS BETWEEN 19 PRECEDING AND CURRENT ROW),
           w60 AS (PARTITION BY ticker ORDER BY date ROWS BETWEEN 59 PRECEDING AND CURRENT ROW)
)
SELECT ticker, date, close,
    ROUND(annualized_vol_20d, 6) AS annualized_vol_20d,
    ROUND(annualized_vol_60d, 6) AS annualized_vol_60d,
    ROUND(annualized_return_20d, 6) AS annualized_return_20d
FROM vol
WHERE annualized_vol_20d IS NOT NULL
ORDER BY ticker, date;

-- fact_momentum_signals: multi-period momentum + MA deviation
CREATE OR REPLACE TABLE fact_momentum_signals AS
WITH price_analytics AS (
    SELECT
        ticker, date, close,
        LAG(close, 5) OVER w AS lag5,
        LAG(close, 20) OVER w AS lag20,
        LAG(close, 60) OVER w AS lag60,
        AVG(close) OVER w20 AS ma20,
        AVG(close) OVER w60 AS ma60
    FROM fact_daily_price
    WINDOW w AS (PARTITION BY ticker ORDER BY date),
           w20 AS (PARTITION BY ticker ORDER BY date ROWS BETWEEN 19 PRECEDING AND CURRENT ROW),
           w60 AS (PARTITION BY ticker ORDER BY date ROWS BETWEEN 59 PRECEDING AND CURRENT ROW)
)
SELECT
    ticker, date, close,
    ROUND((close - lag5) / NULLIF(lag5, 0), 6) AS momentum_5d,
    ROUND((close - lag20) / NULLIF(lag20, 0), 6) AS momentum_20d,
    ROUND((close - lag60) / NULLIF(lag60, 0), 6) AS momentum_60d,
    ROUND((close - ma20) / NULLIF(ma20, 0), 6) AS dist_pct_from_ma20,
    ROUND((close - ma60) / NULLIF(ma60, 0), 6) AS dist_pct_from_ma60,
    CASE
        WHEN close > ma20 AND close > ma60 THEN 'STRONG_UPTREND'
        WHEN close < ma20 AND close < ma60 THEN 'STRONG_DOWNTREND'
        WHEN close > ma20 THEN 'WEAK_UPTREND'
        WHEN close < ma20 THEN 'WEAK_DOWNTREND'
        ELSE 'NEUTRAL'
    END AS trend_signal
FROM price_analytics
WHERE lag5 IS NOT NULL AND lag20 IS NOT NULL AND lag60 IS NOT NULL
ORDER BY ticker, date;

-- fact_ar1_results: AR(1) OLS regression per ticker window
CREATE OR REPLACE TABLE fact_ar1_results AS
WITH return_series AS (
    SELECT
        ticker, date, close,
        (close - LAG(close, 1) OVER (PARTITION BY ticker ORDER BY date))
            / NULLIF(LAG(close, 1) OVER (PARTITION BY ticker ORDER BY date), 0) AS daily_return
    FROM fact_daily_price
),
ar_input AS (
    SELECT
        ticker, date, close, daily_return,
        LAG(daily_return, 1) OVER (PARTITION BY ticker ORDER BY date) AS lag_return
    FROM return_series
    WHERE daily_return IS NOT NULL
),
ar_window AS (
    SELECT
        ticker, date, close, daily_return, lag_return,
        COUNT(*) OVER w AS n_obs,
        SUM(lag_return) OVER w AS sum_lag,
        SUM(daily_return) OVER w AS sum_ret,
        SUM(lag_return * daily_return) OVER w AS sum_lag_ret,
        SUM(lag_return * lag_return) OVER w AS sum_lag_sq,
        CORR(daily_return, lag_return) OVER w AS corr_ar1
    FROM ar_input
    WHERE lag_return IS NOT NULL
    WINDOW w AS (PARTITION BY ticker ORDER BY date ROWS BETWEEN 60 PRECEDING AND 1 PRECEDING)
)
SELECT
    ticker, date, close, daily_return,
    CASE
        WHEN n_obs >= 20 AND (n_obs * sum_lag_sq - sum_lag * sum_lag) != 0
        THEN ROUND(sum_ret * 1.0 / n_obs - ((n_obs * sum_lag_ret - sum_lag * sum_ret) * 1.0
                  / (n_obs * sum_lag_sq - sum_lag * sum_lag)) * (sum_lag * 1.0 / n_obs), 8)
        ELSE NULL
    END AS alpha_ar1,
    CASE
        WHEN n_obs >= 20 AND (n_obs * sum_lag_sq - sum_lag * sum_lag) != 0
        THEN ROUND((n_obs * sum_lag_ret - sum_lag * sum_ret) * 1.0
                  / (n_obs * sum_lag_sq - sum_lag * sum_lag), 8)
        ELSE NULL
    END AS beta_ar1,
    CASE WHEN corr_ar1 IS NOT NULL THEN ROUND(corr_ar1 * corr_ar1, 6) ELSE NULL END AS r_squared_ar1,
    n_obs
FROM ar_window
WHERE n_obs >= 20
ORDER BY ticker, date;
```

```sql
-- ============================================================
-- 7 lightweight OLAP views on top of Star Schema tables

-- v_market_daily_summary
CREATE OR REPLACE VIEW v_market_daily_summary AS
SELECT
    date AS trade_date,
    COUNT(DISTINCT ticker) AS number_of_tickers,
    ROUND(AVG(close), 4) AS avg_close,
    ROUND(AVG(daily_return), 6) AS avg_return,
    SUM(volume) AS total_volume
FROM fact_daily_price
GROUP BY date
ORDER BY date;

-- v_ticker_profile
CREATE OR REPLACE VIEW v_ticker_profile AS
WITH latest_price AS (
    SELECT
        ticker,
        date AS latest_trade_date,
        close AS latest_close,
        volume AS latest_volume,
        ROW_NUMBER() OVER (PARTITION BY ticker ORDER BY date DESC) AS rn
    FROM fact_daily_price
),
ticker_dims AS (
    SELECT
        ticker,
        MAX(company_name) AS company_name,
        MAX(sector) AS sector
    FROM dim_ticker
    WHERE is_current = True
    GROUP BY ticker
)
SELECT
    lp.ticker,
    td.company_name,
    td.sector,
    lp.latest_close,
    lp.latest_volume,
    lp.latest_trade_date
FROM latest_price lp
LEFT JOIN ticker_dims td ON lp.ticker = td.ticker
WHERE lp.rn = 1
ORDER BY lp.ticker;

-- v_fundamental_snapshot (AS OF latest period_date)
CREATE OR REPLACE VIEW v_fundamental_snapshot AS
WITH latest_fundamentals AS (
    SELECT
        ticker,
        fiscal_date,
        period_date,
        report_type,
        revenue,
        net_income,
        total_assets,
        total_liabilities,
        ROW_NUMBER() OVER (PARTITION BY ticker ORDER BY period_date DESC) AS rn
    FROM fact_quarterly_fundamentals
)
SELECT
    ticker,
    fiscal_date,
    MAX(period_date) AS latest_report_date,
    MAX(revenue) AS revenue,
    MAX(net_income) AS net_income,
    MAX(total_assets) AS assets,
    MAX(total_liabilities) AS liabilities
FROM latest_fundamentals
WHERE rn = 1
GROUP BY ticker, fiscal_date
ORDER BY ticker;

-- v_fundamental_history (Bloomberg-style, AS OF join)
CREATE OR REPLACE VIEW v_fundamental_history AS
SELECT
    p.ticker,
    p.date AS price_date,
    f.fiscal_date,
    f.report_type,
    f.freq,
    f.revenue,
    f.net_income,
    f.total_assets,
    f.total_liabilities
FROM fact_daily_price p
LEFT JOIN (
    SELECT
        ticker,
        fiscal_date,
        report_type,
        freq,
        revenue,
        net_income,
        total_assets,
        total_liabilities,
        period_date
    FROM fact_quarterly_fundamentals
) f
    ON p.ticker = f.ticker
    AND f.period_date = (
        SELECT MAX(period_date)
        FROM fact_quarterly_fundamentals f2
        WHERE f2.ticker = p.ticker
          AND f2.period_date <= p.date  -- AS OF join: only past financials
    )
ORDER BY p.ticker, p.date, f.fiscal_date;

-- v_sentiment_price_view
CREATE OR REPLACE VIEW v_sentiment_price_view AS
SELECT
    ticker,
    event_date AS transcript_date,
    sentiment_polarity AS sentiment_score,
    close_on_event,
    next_1d_return,
    next_5d_return
FROM fact_earnings_transcript
ORDER BY ticker, event_date;

-- v_sentiment_binned_returns
CREATE OR REPLACE VIEW v_sentiment_binned_returns AS
SELECT
    CASE
        WHEN sentiment_polarity > 0.2 THEN 'POSITIVE'
        WHEN sentiment_polarity < -0.2 THEN 'NEGATIVE'
        ELSE 'NEUTRAL'
    END AS sentiment_bucket,
    COUNT(*) AS transcript_count,
    ROUND(AVG(next_1d_return), 6) AS avg_1d_return,
    ROUND(AVG(next_5d_return), 6) AS avg_5d_return,
    ROUND(STDDEV(next_1d_return), 6) AS std_1d_return,
    ROUND(AVG(sentiment_subjectivity), 4) AS avg_subjectivity
FROM fact_earnings_transcript
WHERE sentiment_polarity IS NOT NULL
GROUP BY sentiment_bucket;

-- v_sector_rotation
CREATE OR REPLACE VIEW v_sector_rotation AS
WITH sector_daily AS (
    SELECT
        p.date,
        td.sector,
        EXTRACT(YEAR FROM p.date) AS year,
        EXTRACT(QUARTER FROM p.date) AS quarter,
        AVG(p.close) AS avg_close,
        SUM(p.volume) AS total_volume,
        STDDEV(p.close) AS price_std,
        COUNT(DISTINCT p.ticker) AS ticker_count
    FROM fact_daily_price p
    LEFT JOIN dim_ticker td
        ON p.ticker = td.ticker AND td.is_current = True
    GROUP BY p.date, td.sector, year, quarter
),
sector_quarterly AS (
    SELECT
        sector,
        year,
        quarter,
        AVG(avg_close) AS avg_close,
        SUM(total_volume) AS total_volume,
        AVG(price_std) AS avg_volatility,
        AVG(ticker_count) AS avg_ticker_count,
        LAG(AVG(avg_close)) OVER (PARTITION BY sector ORDER BY year, quarter) AS prev_avg_close
    FROM sector_daily
    GROUP BY sector, year, quarter
)
SELECT
    sector,
    year,
    quarter,
    avg_close,
    total_volume,
    avg_volatility,
    avg_ticker_count,
    CASE
        WHEN prev_avg_close IS NOT NULL AND prev_avg_close != 0
        THEN (avg_close - prev_avg_close) / prev_avg_close
        ELSE NULL
    END AS qoq_return,
    RANK() OVER (PARTITION BY year, quarter ORDER BY
        CASE WHEN prev_avg_close IS NOT NULL AND prev_avg_close != 0
        THEN (avg_close - prev_avg_close) / prev_avg_close
        ELSE NULL END DESC NULLS LAST) AS momentum_rank
FROM sector_quarterly
ORDER BY year, quarter, momentum_rank;
```

```python
"""DuckDB connection wrapper for Gold layer queries."""
import duckdb
import pandas as pd
from pathlib import Path

class GoldDataProvider:
    def __init__(self, db_path: str = "duckdb/spx_analytics.duckdb"):
        self._db_path = Path(db_path)
        self._conn = None

    def __enter__(self):
        self._conn = duckdb.connect(str(self._db_path))
        return self

    def __exit__(self, *args):
        if self._conn:
            self._conn.close()

    def execute(self, query: str) -> pd.DataFrame:
        return self._conn.execute(query).fetchdf()
```
